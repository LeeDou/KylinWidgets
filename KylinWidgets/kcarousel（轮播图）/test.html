var carousel = Kcarousel(myCarousel);
carousel.init({
	"width": '600px', // 宽度
	"height": '400px', // 高度
	"dur": 1000, // 每个动画持续时间
	"delay": 5000, // 到下一个动画的延迟时间
	"needOpacity": false // 是否需要透明度变化
}, {
	"leftBtn": leftBtn, // 向左按钮id
	"rightBtn": rightBtn, // 向右按钮id
	"footer": footer // 当前索引展示块id
});

press.onclick = function() {
	alert(carousel.getIndex());
}
