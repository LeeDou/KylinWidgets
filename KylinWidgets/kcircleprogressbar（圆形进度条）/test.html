演示 kcircleprogressbar（圆形进度条）

组件结构以及初始化





	

	

		
		
	






	


					
var bar = Kcircleprogressbar(myBar);
bar.init({
	'radius': '150px', //半径
	'thick': '45px', //厚度
	'bgColor': '#59f', //背景颜色
	'fillColor': '#eee', //空白填充颜色
	'proColor': '#F2E42C', //进度颜色
	'textColor': 'orange', //文字颜色
	'textSize': '50px' //文字大小
}, {
	'process': '0%' //当前进度
});
function addPro() {
	var v = bar.getProcess();
	v++;
	bar.setProcess(v);
}
add.onclick = function() {
	addPro();
}
sub.onclick = function() {
	var v = bar.getProcess();
	v = v - 10;
	bar.setProcess(v);
}
setInterval(function() {
	addPro();
}, 50);
