演示 klineprogressbar（条形进度条）

组件结构以及初始化




	

	

	

		
		
	





	





	



				

var bar = Klineprogressbar(myBar);
bar.init({
	'width': '600px',
	'height': '15px',
	'proColor': 'orange',
	'bgColor': '#aaa',
	'isShowProgress': true
}, {
	'maxValue': 100,
	'minValue': -100,
	'value': -80
});
function showInfo() {
	show.innerHTML = bar.getValue() + '/(' + bar.getMaxValue() + ',' + bar.getMinValue() + ')/' + bar.getProcess();
}
showInfo();
function addPro() {
	var v = bar.getValue();
	v++;
	bar.setValue(v);
	showInfo();
}
add.onclick = function() {
	addPro();
}
sub.onclick = function() {
	var v = bar.getValue();
	v = v - 10;
	bar.setValue(v);
	showInfo();
}
setInterval(function() {
	addPro();
}, 50);
