演示 kloading（加载图）

组件结构以及初始化




	






	



				

var loading = document.getElementById("myLoading");

Kloading(loading).init({
	'radius': '60px', // 半径
	'thick': '10px', // 厚度
	'color': 'orange', // 颜色
	'bgColor': '#59f', // 背景颜色
	'time': 1200, // 循环一个周期所用时间
	'model': 'model_1' // 加载器模式选择
});
