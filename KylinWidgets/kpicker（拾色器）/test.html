演示 kpicker（拾色器）

组件结构以及初始化






	


				

var pickerTarget = Kpicker(mypicker);
pickerTarget.init({
	'surfaceColor': '#59f' // 外观颜色
});

press.onclick = function() {
	var color1 = pickerTarget.getColor();
	var color2 = pickerTarget.getColor('rgb');
	var color3 = pickerTarget.getColor('hex');
	alert(color1 + '\n' + color2 + '\n' + color3);
};
