演示 kslidebar（滑动条）

组件结构以及初始化




	






	



				

Kslidebar(myBar).init({
	'width': '450px', // 进度条宽度
	'height': '20px', // 进度条高度
	'isFillet': true, // 是否使用圆角
	'proColor': '#59f', // 进度颜色
	'bgColor': '#ccc', // 进度条背景颜色
	'cIsArc': true, // 控制块是否使用弧形样式
	'cwidth': '15px', // 控制块宽度
	'ccolor': '#888', // 控制块颜色
	'isShowProgress': true, // 是否显示进度提示文字
	'textColor': '#fff' // 进度提示文字颜色
}, {
	'maxValue': 100, // 最大值
	'minValue': 0, // 最小值
	'value': 30 // 初始化值
});
