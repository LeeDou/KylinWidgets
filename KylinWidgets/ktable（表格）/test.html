var tab = document.getElementById("tab");
var targetTab = Ktable(tab);

drow.onclick = function() {
	targetTab.deleteRow(0); // 删除第0行
}
dcol.onclick = function() {
	targetTab.deleteCol(0); // 删除第0列
}

irow.onclick = function() {
	targetTab.insertRow(-1); // 从尾部插入一个空行
}

icol.onclick = function() {
	targetTab.insertCol(-1); // 从尾部插入一个空列
}

pdata.onclick = function() {
	alert(targetTab.getDataList()); // 打印表中所有数据
}
